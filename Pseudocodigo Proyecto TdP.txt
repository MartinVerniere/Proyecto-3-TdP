Pseudocodigo generacion Mapa:
Hilo mapa{
	mientras (dos tandas no esten vacias) {
		cada X segundos y la (tanda no se termino){
			obtengo el siguiente infectado de la tanda;
			la posicion inicial del infectado es (rnd=x, tope del mapa);
		}
		si (tanda esta vacia y no es la ultima), cambio a la tanda siguiente;
}


Pseudocodigo movimiento:

Hilo movimientos{
	cada X segundos{
		para cada objeto de la pantalla que no sea el jugador{
			mover los sprites hacia abajo en el eje vertical del mapa (moverlos en la cantidad igual a su velocidad);
			si (infectado o particula colisiona con jugador){
				jugador.colision(visitor_infectado) o jugador.colision(visitor_particula)
			}
			si llego al final del mapa {
				si (es un infectado){
					mover al infectado al tope del mapa, en la misma posicion del eje horizontal;
				}
				si no{
					eliminarlo del juego;
				}
			}
			para cada particula en la pantalla{
				aumentar la distancia recorrida;
				si (distanca recorrida > alcancemaximo){
					eliminar particula
				}
			}
			si (infectado desprende particula){
				crear sprite particula
				crear particula(sprite particula);
			}
		}
	}
}
Hilo disparo{
	espera X segundos;
	borra el sprite disparo;
}
